Airport Group Examples

Here are several examples, which you can copy and paste into the airport groups app.

Example 1

g1 = {ICAO:KSFO};
print(g1);

Example 2

g1 = {ICAO:KMDW};
print(g1);
g2 = {FAA:LAX, FAA:PAO, FAA:SFO};
print(g2);
g3 = g1 + g2;
print(g3);
g4 = g2 ^ {FAA:PAO, FAA:SFO, FAA:STL};
print(g4);
g5 = {ICAO:KSFO} + g1;
print(g5);

Example 3

f1 = isInCountry(USA);
g1 = {FAA:SFO, ICAO:SAEZ, ICAO:EGLL};
print(g1);
g2 = g1 | f1;
print(g2);
f2 = isInCountry(ARG);
g3 = g1 | f2;
print(g3);
g4 = g1 | isInCountry(ARG);
print(g4);
g5 = g1 | isInCountry(GBR);
print(g5);
f3 = isInCountry(ARG) || isInCountry(GBR);
g6 = g1 | f3;
print(g6);
g7 = g1 | !f3;
print(g7);
g8 = g1 | ! isInCountry(USA);
print(g8);

Example 4

g1 = {FAA:SFO};
g2 = {FAA:LAX};
g3 = {FAA:MIA, FAA:LAX};
g4 = (g1 + g2) ^ g3;
print(g4);
g5 = g1 + (g2 ^ g3);
print(g5);

Example 5

g1 = {FAA:SFO, FAA:LAX, FAA:MIA};
f1 = isInCountry(USA);
f2 = isInState(CA);
f3 = isNorthOfLatitude(30);
g2 = g1 | f3;
print(g2);

Example 6

g1 = {FAA:SEA, FAA:SFO, FAA:LAX, FAA:DEN, FAA:DAL, FAA:MDW, FAA:TEB, FAA:MIA};
f1 = isNorthOf(36) && isSouthOf(42);
g2 = g1 | f1;
print(g2);
f2 = isEastOf(-110) && isWestOf(-90);
g3 = g1 | f1 && f2;
print(g3);
f3 = f1 && (isEastOf(-110) && isWestOf(-90));
g4 = g1 | f3;
print(g4);
g5 = g1 | (isNorthOf(36) && isSouthOf(42)) && (isEastOf(-110) && isWestOf(-90));
print(g5);
g6 = g1 | !isSouthOf(45);
print(g6);
g7 = g1 | (isEastOf(-75) || isWestOf(-120));
print(g7);

Example 7

g1 = {FAA:SEA, FAA:SFO, FAA:LAX, FAA:DEN, FAA:DAL, FAA:MDW, FAA:TEB, FAA:MIA};
airportIsInSet(FAA:SEA, g1);
airportIsInSet(ICAO:KSEA, g1);
airportIsInSet(FAA:EQY, g1);

Example 8

g1 = {FAA:SEA, FAA:SFO, FAA:SJC, FAA:OAK, FAA:PAO, FAA:LAX, FAA:DEN, FAA:DAL, FAA:MDW, FAA:TEB, FAA:MIA, FAA:SQL};
g2 = g1 | isNearAirport(FAA:SFO, 50);
print(g2);

Example 9

g1 = allAirports | isInCountry(USA);
g2 = g1 | isInState(RI);
print(g2);
g3 = g1 | isNearAirport(FAA:SFO, 50);
print(g3);

Example 10

g1 = {FAA:SFO, FAA:LAX, FAA:SJC, FAA:TEB};
print(g1);
g2 = {FAA:JFK, FAA:SFO, FAA:LAX, FAA:MIA, FAA:MDW};
print(g2);
g3 = g1 - g2;
print(g3);
g4 = g2 - g1;
print(g4);
